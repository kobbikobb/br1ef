44/// Convert the input into what [the IMAP
45/// grammar](https://tools.ietf.org/html/rfc3501#section-9)
46/// calls "quoted", which is reachable from "string" et al.
47/// Also ensure it doesn't contain a colliding command-delimiter (newline).
48fn validate_str(value: &str) -> Result<String> {
49    validate_str_noquote(value)?;
50    Ok(quote!(value))
51}
52
53/// Ensure the input doesn't contain a command-terminator (newline), but don't quote it like
54/// `validate_str`.
55/// This is helpful for things like the FETCH attributes, which,
56/// per [the IMAP grammar](https://tools.ietf.org/html/rfc3501#section-9) may not be quoted:
57///
58/// > fetch     = "FETCH" SP sequence-set SP ("ALL" / "FULL" / "FAST" /
59/// >             fetch-att / "(" fetch-att *(SP fetch-att) ")")
60/// >
61/// > fetch-att = "ENVELOPE" / "FLAGS" / "INTERNALDATE" /
62/// >             "RFC822" [".HEADER" / ".SIZE" / ".TEXT"] /
63/// >             "BODY" ["STRUCTURE"] / "UID" /
64/// >             "BODY" section ["<" number "." nz-number ">"] /
65/// >             "BODY.PEEK" section ["<" number "." nz-number ">"]
66///
67/// Note the lack of reference to any of the string-like rules or the quote characters themselves.
68fn validate_str_noquote(value: &str) -> Result<&str> {
69    value
70        .matches(|c| c == '\n' || c == '\r')
71        .next()
72        .and_then(|s| s.chars().next())
73        .map(|offender| Error::Validate(ValidateError(offender)))
74        .err()?;
75    Ok(value)
76}

78/// This ensures the input doesn't contain a command-terminator or any other whitespace
79/// while leaving it not-quoted.
80/// This is needed because, per [the formal grammer given in RFC
81/// 3501](https://tools.ietf.org/html/rfc3501#section-9), a sequence set consists of the following:
82///
83/// > sequence-set = (seq-number / seq-range) *("," sequence-set)
84/// > seq-range = seq-number ":" seq-number
85/// > seq-number = nz-number / "*"
86/// > nz-number       = digit-nz *DIGIT
87/// > digit-nz        = %x31-39
88///
89/// Note the lack of reference to SP or any other such whitespace terminals.
90/// Per this grammar, in theory we ought to be even more restrictive than "no whitespace".
91fn validate_sequence_set(value: &str) -> Result<&str> {
92    value
93        .matches(|c: char| c.is_ascii_whitespace())
94        .next()
95        .and_then(|s| s.chars().next())
96        .map(|offender| Error::Validate(ValidateError(offender)))
97        .err()?;
98    Ok(value)
99}

176/// Connect to a server using a TLS-encrypted connection.
177///
178/// The returned [`Client`] is unauthenticated; to access session-related methods (through
179/// [`Session`]), use [`Client::login`] or [`Client::authenticate`].
180///
181/// The domain must be passed in separately from the `TlsConnector` so that the certificate of the
182/// IMAP server can be validated.
183///
184/// # Examples
185///
186/// ```no_run
187/// # extern crate native_tls;
188/// # extern crate imap;
189/// # use std::io;
190/// # use native_tls::TlsConnector;
191/// # fn main() {
192/// let tls = TlsConnector::builder().build().unwrap();
193/// let client = imap::connect(("imap.example.org", 993), "imap.example.org", &tls).unwrap();
194/// # }
195/// ```
196#[cfg(feature = "tls")]
197pub fn connect<A: ToSocketAddrs, S: AsRef<str>>(
198    addr: A,
199    domain: S,
200    ssl_connector: &TlsConnector,
201) -> Result<Client<TlsStream<TcpStream>>> {
202    match TcpStream::connect(addr) {
203        Ok(stream) => {
204            let ssl_stream = match TlsConnector::connect(ssl_connector, domain.as_ref(), stream) {
205                Ok(s) => s,
206                Err(e) => return Err(Error::TlsHandshake(e)),
207            };
208            let mut socket = Client::new(ssl_stream);
209
210            socket.read_greeting()?;
211            Ok(socket)
212        }
213        Err(e) => Err(Error::Io(e)),
214    }
215}

217/// Connect to a server and upgrade to a TLS-encrypted connection.
218///
219/// This is the [STARTTLS](https://tools.ietf.org/html/rfc2595) equivalent to [`connect`]. All
220/// notes there also apply here.
221///
222/// # Examples
223///
224/// ```no_run
225/// # extern crate native_tls;
226/// # extern crate imap;
227/// # use std::io;
228/// # use native_tls::TlsConnector;
229/// # fn main() {
230/// let tls = TlsConnector::builder().build().unwrap();
231/// let client = imap::connect_starttls(("imap.example.org", 143), "imap.example.org", &tls).unwrap();
232/// # }
233/// ```
234#[cfg(feature = "tls")]
235pub fn connect_starttls<A: ToSocketAddrs, S: AsRef<str>>(
236    addr: A,
237    domain: S,
238    ssl_connector: &TlsConnector,
239) -> Result<Client<TlsStream<TcpStream>>> {
240    match TcpStream::connect(addr) {
241        Ok(stream) => {
242            let mut socket = Client::new(stream);
243            socket.read_greeting()?;
244            socket.run_command_and_check_ok("STARTTLS")?;
245            TlsConnector::connect(
246                ssl_connector,
247                domain.as_ref(),
248                socket.conn.stream.into_inner()?,
249            )
250            .map(Client::new)
251            .map_err(Error::TlsHandshake)
252        }
253        Err(e) => Err(Error::Io(e)),
254    }
255}

279// As the pattern of returning the unauthenticated `Client` (a.k.a. `self`) back with a login error
280// is relatively common, it's abstacted away into a macro here.
281//
282// Note: 1) using `.map_err(|e| (e, self))` or similar here makes the closure own self, so we can't
283//          do that.
284//       2) in theory we wouldn't need the second parameter, and could just use the identifier
285//          `self` from the surrounding function, but being explicit here seems a lot cleaner.
286macro_rules! ok_or_unauth_client_err {
287    ($r:expr, $self:expr) => {
288        match $r {
289            Ok(o) => o,
290            Err(e) => return Err((e, $self)),
291        }
292    };
293}

338    /// returned; on error the original `Client` instance is returned in addition to the error.
339    /// This is because `login` takes ownership of `self`, so in order to try again (e.g. after
340    /// prompting the user for credetials), ownership of the original `Client` needs to be

350    /// let client = imap::connect(
351    ///     ("imap.example.org", 993),
352    ///     "imap.example.org",

356    ///     Ok(s) => {
357    ///         // you are successfully authenticated!
358    ///     },
359    ///     Err((e, orig_client)) => {
360    ///         eprintln!("error logging in: {}", e);
361    ///         // prompt user and try again with orig_client here

369        username: U,
370        password: P,
371    ) -> ::std::result::Result<Session<T>, (Error, Client<T>)> {
372        let u = ok_or_unauth_client_err!(validate_str(username.as_ref()), self);
373        let p = ok_or_unauth_client_err!(validate_str(password.as_ref()), self);
374        ok_or_unauth_client_err!(
375            self.run_command_and_check_ok(&format!("LOGIN {} {}", u, p)),
376            self
377        );
378
379        Ok(Session::new(self.conn))

383    /// challenge.
384    ///
385    /// ```no_run
386    /// extern crate imap;
387    /// extern crate native_tls;
388    /// use native_tls::TlsConnector;
389    ///
390    /// struct OAuth2 {
391    ///     user: String,
392    ///     access_token: String,
393    /// }
394    ///
395    /// impl imap::Authenticator for OAuth2 {
396    ///     type Response = String;
397    ///     fn process(&self, _: &[u8]) -> Self::Response {
398    ///         format!(
399    ///             "user={}\x01auth=Bearer {}\x01\x01",
400    ///             self.user, self.access_token
401    ///         )
402    ///     }
403    /// }
404    ///
405    /// fn main() {
406    ///     let auth = OAuth2 {
407    ///         user: String::from("me@example.com"),
408    ///         access_token: String::from("<access_token>"),
409    ///     };
410    ///     let domain = "imap.example.com";
411    ///     let tls = TlsConnector::builder().build().unwrap();
412    ///     let client = imap::connect((domain, 993), domain, &tls).unwrap();
413    ///     match client.authenticate("XOAUTH2", &auth) {
414    ///         Ok(session) => {
415    ///             // you are successfully authenticated!
416    ///         },
417    ///         Err((e, orig_client)) => {
418    ///             eprintln!("error authenticating: {}", e);
419    ///             // prompt user and try again with orig_client here
420    ///             return;
421    ///         }
422    ///     };
423    /// }
424    /// ```
425    pub fn authenticate<A: Authenticator, S: AsRef<str>>(
426        mut self,
427        auth_type: S,
428        authenticator: &A,
429    ) -> ::std::result::Result<Session<T>, (Error, Client<T>)> {
430        ok_or_unauth_client_err!(
431            self.run_command(&format!("AUTHENTICATE {}", auth_type.as_ref())),
432            self
433        );
434        self.do_auth_handshake(authenticator)
435    }

437    /// This func does the handshake process once the authenticate command is made.
438    fn do_auth_handshake<A: Authenticator>(
439        mut self,
440        authenticator: &A,
441    ) -> ::std::result::Result<Session<T>, (Error, Client<T>)> {
442        // TODO Clean up this code
443        loop {
444            let mut line = Vec::new();
445
446            // explicit match blocks neccessary to convert error to tuple and not bind self too
447            // early (see also comment on `login`)
448            ok_or_unauth_client_err!(self.readline(&mut line), self);
449
450            // ignore server comments
451            if line.starts_with(b"* ") {
452                continue;
453            }
454
455            // Some servers will only send `+\r\n`.
456            if line.starts_with(b"+ ") || &line == b"+\r\n" {
457                let challenge = if &line == b"+\r\n" {
458                    Vec::new()
459                } else {
460                    let line_str = ok_or_unauth_client_err!(
461                        match str::from_utf8(line.as_slice()) {
462                            Ok(line_str) => Ok(line_str),
463                            Err(e) => Err(Error::Parse(ParseError::DataNotUtf8(line, e))),
464                        },
465                        self
466                    );
467                    let data =
468                        ok_or_unauth_client_err!(parse_authenticate_response(line_str), self);
469                    ok_or_unauth_client_err!(
470                        base64::decode(data).map_err(|e| Error::Parse(ParseError::Authentication(
471                            data.to_string(),
472                            Some(e)
473                        ))),
474                        self
475                    )
476                };
477
478                let raw_response = &authenticator.process(&challenge);
479                let auth_response = base64::encode(raw_response);
480                ok_or_unauth_client_err!(
481                    self.write_line(auth_response.into_bytes().as_slice()),
482                    self
483                );
484            } else {
485                ok_or_unauth_client_err!(self.read_response_onto(&mut line), self);
486                return Ok(Session::new(self.conn));
487            }
488        }
489    }

503    /// Selects a mailbox
504    ///
505    /// The `SELECT` command selects a mailbox so that messages in the mailbox can be accessed.
506    /// Note that earlier versions of this protocol only required the FLAGS, EXISTS, and RECENT
507    /// untagged data; consequently, client implementations SHOULD implement default behavior for
508    /// missing data as discussed with the individual item.
509    ///
510    /// Only one mailbox can be selected at a time in a connection; simultaneous access to multiple
511    /// mailboxes requires multiple connections.  The `SELECT` command automatically deselects any
512    /// currently selected mailbox before attempting the new selection. Consequently, if a mailbox
513    /// is selected and a `SELECT` command that fails is attempted, no mailbox is selected.
514    ///
515    /// Note that the server *is* allowed to unilaterally send things to the client for messages in
516    /// a selected mailbox whose status has changed. See the note on [unilateral server responses
517    /// in RFC 3501](https://tools.ietf.org/html/rfc3501#section-7). This means that if you use
518    /// [`Connection::run_command_and_read_response`], you *may* see additional untagged `RECENT`,
519    /// `EXISTS`, `FETCH`, and `EXPUNGE` responses. You can get them from the
520    /// `unsolicited_responses` channel of the [`Session`](struct.Session.html).
521    pub fn select<S: AsRef<str>>(&mut self, mailbox_name: S) -> Result<Mailbox> {
522        // TODO: also note READ/WRITE vs READ-only mode!
523        self.run_command_and_read_response(&format!(
524            "SELECT {}",
525            validate_str(mailbox_name.as_ref())?
526        ))
527        .and_then(|lines| parse_mailbox(&lines[..], &mut self.unsolicited_responses_tx))
528    }

530    /// The `EXAMINE` command is identical to [`Session::select`] and returns the same output;
531    /// however, the selected mailbox is identified as read-only. No changes to the permanent state
532    /// of the mailbox, including per-user state, will happen in a mailbox opened with `examine`;
533    /// in particular, messagess cannot lose [`Flag::Recent`] in an examined mailbox.
534    pub fn examine<S: AsRef<str>>(&mut self, mailbox_name: S) -> Result<Mailbox> {

538        ))

540    }

542    /// Fetch retrieves data associated with a set of messages in the mailbox.
543    ///
544    /// Note that the server *is* allowed to unilaterally include `FETCH` responses for other
545    /// messages in the selected mailbox whose status has changed. See the note on [unilateral
546    /// server responses in RFC 3501](https://tools.ietf.org/html/rfc3501#section-7).
547    ///
548    /// `query` is a list of "data items" (space-separated in parentheses if `>1`). There are three
549    /// "macro items" which specify commonly-used sets of data items, and can be used instead of
550    /// data items.  A macro must be used by itself, and not in conjunction with other macros or
551    /// data items. They are:
552    ///
553    ///  - `ALL`: equivalent to: `(FLAGS INTERNALDATE RFC822.SIZE ENVELOPE)`
554    ///  - `FAST`: equivalent to: `(FLAGS INTERNALDATE RFC822.SIZE)`
555    ///
556    /// The currently defined data items that can be fetched are listen [in the
557    /// RFC](https://tools.ietf.org/html/rfc3501#section-6.4.5), but here are some common ones:
558    ///
559    ///  - `FLAGS`: The flags that are set for this message.
560    ///  - `INTERNALDATE`: The internal date of the message.
561    ///  - `BODY[<section>]`:
562    ///
563    ///     The text of a particular body section.  The section specification is a set of zero or
564    ///     more part specifiers delimited by periods.  A part specifier is either a part number
565    ///     (see RFC) or one of the following: `HEADER`, `HEADER.FIELDS`, `HEADER.FIELDS.NOT`,
566    ///     `MIME`, and `TEXT`.  An empty section specification (i.e., `BODY[]`) refers to the
567    ///     entire message, including the header.
568    ///
569    ///     The `HEADER`, `HEADER.FIELDS`, and `HEADER.FIELDS.NOT` part specifiers refer to the
570    ///     [RFC-2822](https://tools.ietf.org/html/rfc2822) header of the message or of an
571    ///     encapsulated [MIME-IMT](https://tools.ietf.org/html/rfc2046)
572    ///     MESSAGE/[RFC822](https://tools.ietf.org/html/rfc822) message. `HEADER.FIELDS` and
573    ///     `HEADER.FIELDS.NOT` are followed by a list of field-name (as defined in
574    ///     [RFC-2822](https://tools.ietf.org/html/rfc2822)) names, and return a subset of the
575    ///     header.  The subset returned by `HEADER.FIELDS` contains only those header fields with
576    ///     a field-name that matches one of the names in the list; similarly, the subset returned
577    ///     by `HEADER.FIELDS.NOT` contains only the header fields with a non-matching field-name.
578    ///     The field-matching is case-insensitive but otherwise exact.  Subsetting does not
579    ///     exclude the [RFC-2822](https://tools.ietf.org/html/rfc2822) delimiting blank line
580    ///     between the header and the body; the blank line is included in all header fetches,
581    ///     except in the case of a message which has no body and no blank line.
582    ///
583    ///     The `MIME` part specifier refers to the [MIME-IMB](https://tools.ietf.org/html/rfc2045)
584    ///     header for this part.
585    ///
586    ///     The `TEXT` part specifier refers to the text body of the message,

603        S2: AsRef<str>,

607        } else {

609                "FETCH {} {}",
610                validate_sequence_set(sequence_set.as_ref())?,
611                validate_str_noquote(query.as_ref())?
612            ))
613            .and_then(|lines| parse_fetches(lines, &mut self.unsolicited_responses_tx))
614        }
615    }

1060    /// The [`STATUS` command](https://tools.ietf.org/html/rfc3501#section-6.3.10) requests the
1061    /// status of the indicated mailbox. It does not change the currently selected mailbox, nor
1062    /// does it affect the state of any messages in the queried mailbox (in particular, `status`
1063    /// will not cause messages to lose [`Flag::Recent`]).
1064    ///
1065    /// `status` provides an alternative to opening a second [`Session`] and using
1066    /// [`Session::examine`] on a mailbox to query that mailbox's status without deselecting the
1067    /// current mailbox in the first `Session`.
1068    ///
1069    /// Unlike [`Session::list`], `status` is not guaranteed to be fast in its response.  Under
1070    /// certain circumstances, it can be quite slow.  In some implementations, the server is
1071    /// obliged to open the mailbox read-only internally to obtain certain status information.
1072    /// Also unlike [`Session::list`], `status` does not accept wildcards.
1073    ///
1074    /// > Note: `status` is intended to access the status of mailboxes other than the currently
1075    /// > selected mailbox.  Because `status` can cause the mailbox to be opened internally, and
1076    /// > because this information is available by other means on the selected mailbox, `status`
1077    /// > SHOULD NOT be used on the currently selected mailbox.
1078    ///
1079    /// The STATUS command MUST NOT be used as a "check for new messages in the selected mailbox"
1080    /// operation (refer to sections [7](https://tools.ietf.org/html/rfc3501#section-7),
1081    /// [7.3.1](https://tools.ietf.org/html/rfc3501#section-7.3.1), and
1082    /// [7.3.2](https://tools.ietf.org/html/rfc3501#section-7.3.2) for more information about the
1083    /// proper method for new message checking).
1084    ///
1085    /// The currently defined status data items that can be requested are:
1086    ///
1087    ///  - `MESSAGES`: The number of messages in the mailbox.
1088    ///  - `RECENT`: The number of messages with [`Flag::Recent`] set.
1089    ///  - `UIDNEXT`: The next [`Uid`] of the mailbox.
1090    ///  - `UIDVALIDITY`: The unique identifier validity value of the mailbox (see [`Uid`]).
1091    ///  - `UNSEEN`: The number of messages which do not have [`Flag::Seen`] set.
1092    ///
1093    /// `data_items` is a space-separated list enclosed in parentheses.
1094    pub fn status<S1: AsRef<str>, S2: AsRef<str>>(
1095        &mut self,
1096        mailbox_name: S1,
1097        data_items: S2,
1098    ) -> Result<Mailbox> {
1099        self.run_command_and_read_response(&format!(
1100            "STATUS {} {}",
1101            validate_str(mailbox_name.as_ref())?,
1102            data_items.as_ref()
1103        ))
1104        .and_then(|lines| parse_mailbox(&lines[..], &mut self.unsolicited_responses_tx))
1105    }

1130    /// `content` as a new message to the end of the specified destination `mailbox`.  This
1131    /// argument SHOULD be in the format of an [RFC-2822](https://tools.ietf.org/html/rfc2822)

1135    /// > lines are omitted in the message literal argument to `append`.  The full implications of
1136    /// > doing so MUST be understood and carefully weighed.
1137    ///
1138    /// If the append is unsuccessful for any reason, the mailbox is restored to its state before
1139    /// the append attempt; no partial appending will happen.
1140    ///

1142    /// automatically create the mailbox.

1146    /// `EXISTS` response.  If the server does not do so, the client MAY issue a `NOOP` command (or
1147    /// failing that, a `CHECK` command) after one or more `APPEND` commands.

1154    ///
1155    /// > If a flag parenthesized list is specified, the flags SHOULD be set

1157    /// > resulting message is set to empty by default.  In either case, the
1158    /// > Recent flag is also set.

1171    /// The [`APPEND` command](https://tools.ietf.org/html/rfc3501#section-6.3.11) can take
1172    /// an optional FLAGS parameter to set the flags on the new message.
1173    ///
1174    /// > If a flag parenthesized list is specified, the flags SHOULD be set
1175    /// > in the resulting message; otherwise, the flag list of the
1176    /// > resulting message is set to empty by default.  In either case, the
1177    /// > Recent flag is also set.
1178    ///
1179    /// The [`\Recent` flag](https://tools.ietf.org/html/rfc3501#section-2.3.2) is not
1180    /// allowed as an argument to `APPEND` and will be filtered out if present in `flags`.
1181    ///
1182    /// Pass a date in order to set the date that the message was originally sent.
1183    ///
1184    /// > If a date-time is specified, the internal date SHOULD be set in
1185    /// > the resulting message; otherwise, the internal date of the
1186    /// > resulting message is set to the current date and time by default.
1187    pub fn append_with_flags_and_date<S: AsRef<str>, B: AsRef<[u8]>>(
1188        &mut self,
1189        mailbox: S,
1190        content: B,
1191        flags: &[Flag<'_>],
1192        date: impl Into<Option<DateTime<FixedOffset>>>,
1193    ) -> Result<()> {
1194        let content = content.as_ref();
1195        let flagstr = flags
1196            .iter()
1197            .filter(|f| **f != Flag::Recent)
1198            .map(|f| f.to_string())
1199            .collect::<Vec<String>>()
1200            .join(" ");
1201        let datestr = match date.into() {
1202            Some(date) => format!(" \"{}\"", date.format("%d-%h-%Y %T %z")),
1203            None => "".to_string(),
1204        };
1205
1206        self.run_command(&format!(
1207            "APPEND \"{}\" ({}){} {{{}}}",
1208            mailbox.as_ref(),
1209            flagstr,
1210            datestr,
1211            content.len()
1212        ))?;
1213        let mut v = Vec::new();
1214        self.readline(&mut v)?;
1215        if !v.starts_with(b"+") {
1216            return Err(Error::Append);
1217        }
1218        self.stream.write_all(content)?;
1219        self.stream.write_all(b"\r\n")?;
1220        self.stream.flush()?;
1221        self.read_response().map(|_| ())
1222    }

1224    /// The [`SEARCH` command](https://tools.ietf.org/html/rfc3501#section-6.4.4) searches the
1225    /// mailbox for messages that match the given `query`.  `query` consist of one or more search
1226    /// keys separated by spaces.  The response from the server contains a listing of [`Seq`]s
1227    /// corresponding to those messages that match the searching criteria.
1228    ///
1229    /// When multiple search keys are specified, the result is the intersection of all the messages
1230    /// that match those keys.  Or, in other words, only messages that match *all* the keys. For
1231    /// example, the criteria
1232    ///
1233    /// ```text
1234    /// DELETED FROM "SMITH" SINCE 1-Feb-1994
1235    /// ```
1236    ///
1237    /// refers to all deleted messages from Smith that were placed in the mailbox since February 1,
1238    /// 1994.  A search key can also be a parenthesized list of one or more search keys (e.g., for
1239    /// use with the `OR` and `NOT` keys).
1240    ///
1241    /// In all search keys that use strings, a message matches the key if the string is a substring
1242    /// of the field.  The matching is case-insensitive.
1243    ///
1244    /// Below is a selection of common search keys.  The full list can be found in the
1245    /// specification of the [`SEARCH command`](https://tools.ietf.org/html/rfc3501#section-6.4.4).
1246    ///
1247    ///  - `NEW`: Messages that have [`Flag::Recent`] set but not [`Flag::Seen`]. This is functionally equivalent to `(RECENT UNSEEN)`.
1248    ///  - `OLD`: Messages that do not have [`Flag::Recent`] set.  This is functionally equivalent to `NOT RECENT` (as opposed to `NOT NEW`).
1249    ///  - `RECENT`: Messages that have [`Flag::Recent`] set.
1250    ///  - `ANSWERED`: Messages with [`Flag::Answered`] set.
1251    ///  - `DELETED`: Messages with [`Flag::Deleted`] set.
1252    ///  - `DRAFT`: Messages with [`Flag::Draft`] set.
1253    ///  - `FLAGGED`: Messages with [`Flag::Flagged`] set.
1254    ///  - `SEEN`: Messages that have [`Flag::Seen`] set.
1255    ///  - `<sequence set>`: Messages with message sequence numbers corresponding to the specified message sequence number set.
1256    ///  - `UID <sequence set>`: Messages with [`Uid`] corresponding to the specified unique identifier set.  Sequence set ranges are permitted.
1257    ///
1258    ///  - `SUBJECT <string>`: Messages that contain the specified string in the envelope structure's `SUBJECT` field.
1259    ///  - `BODY <string>`: Messages that contain the specified string in the body of the message.
1260    ///  - `FROM <string>`: Messages that contain the specified string in the envelope structure's `FROM` field.
1261    ///  - `TO <string>`: Messages that contain the specified string in the envelope structure's `TO` field.
1262    ///
1263    ///  - `NOT <search-key>`: Messages that do not match the specified search key.
1264    ///  - `OR <search-key1> <search-key2>`: Messages that match either search key.
1265    ///
1266    ///  - `BEFORE <date>`: Messages whose internal date (disregarding time and timezone) is earlier than the specified date.
1267    ///  - `SINCE <date>`: Messages whose internal date (disregarding time and timezone) is within or later than the specified date.
1268    pub fn search<S: AsRef<str>>(&mut self, query: S) -> Result<HashSet<Seq>> {
1269        self.run_command_and_read_response(&format!("SEARCH {}", query.as_ref()))
1270            .and_then(|lines| parse_ids(&lines, &mut self.unsolicited_responses_tx))
1271    }

1341    pub(crate) fn read_response_onto(&mut self, data: &mut Vec<u8>) -> Result<()> {
1342        let mut continue_from = None;
1343        let mut try_first = !data.is_empty();
1344        let match_tag = format!("{}{}", TAG_PREFIX, self.tag);
1345        loop {
1346            let line_start = if try_first {
1347                try_first = false;
1348                0
1349            } else {
1350                let start_new = data.len();
1351                self.readline(data)?;
1352                continue_from.take().unwrap_or(start_new)
1353            };
1354
1355            let break_with = {
1356                use imap_proto::{parse_response, Response, Status};
1357                let line = &data[line_start..];
1358
1359                match parse_response(line) {
1360                    Ok((
1361                        _,
1362                        Response::Done {
1363                            tag,
1364                            status,
1365                            information,
1366                            ..
1367                        },
1368                    )) => {
1369                        assert_eq!(tag.as_bytes(), match_tag.as_bytes());
1370                        Some(match status {
1371                            Status::Bad | Status::No => {
1372                                Err((status, information.map(ToString::to_string)))
1373                            }
1374                            Status::Ok => Ok(()),
1375                            status => Err((status, None)),
1376                        })
1377                    }
1378                    Ok((..)) => None,
1379                    Err(nom::Err::Incomplete(..)) => {
1380                        continue_from = Some(line_start);
1381                        None
1382                    }
1383                    _ => Some(Err((Status::Bye, None))),
1384                }
1385            };
1386
1387            match break_with {
1388                Some(Ok(_)) => {
1389                    data.truncate(line_start);
1390                    break Ok(());
1391                }
1392                Some(Err((status, expl))) => {
1393                    use imap_proto::Status;
1394                    match status {
1395                        Status::Bad => {
1396                            break Err(Error::Bad(
1397                                expl.unwrap_or_else(|| "no explanation given".to_string()),
1398                            ));
1399                        }
1400                        Status::No => {
1401                            break Err(Error::No(
1402                                expl.unwrap_or_else(|| "no explanation given".to_string()),
1403                            ));
1404                        }
1405                        _ => break Err(Error::Parse(ParseError::Invalid(data.split_off(0)))),
1406                    }
1407                }
1408                None => {}
1409            }
1410        }
1411    }

1413    pub(crate) fn readline(&mut self, into: &mut Vec<u8>) -> Result<usize> {
1414        use std::io::BufRead;
1415        let read = self.stream.read_until(LF, into)?;
1416        if read == 0 {
1417            return Err(Error::ConnectionLost);
1418        }
1419
1420        if self.debug {
1421            // Remove CRLF
1422            let len = into.len();
1423            let line = &into[(len - read)..(len - 2)];
1424            eprint!("S: {}\n", String::from_utf8_lossy(line));
1425        }
1426
1427        Ok(read)
1428    }

1431        self.tag += 1;
1432        format!("{}{} {}", TAG_PREFIX, self.tag, command)
1433    }

1435    pub(crate) fn write_line(&mut self, buf: &[u8]) -> Result<()> {
1436        self.stream.write_all(buf)?;
1437        self.stream.write_all(&[CR, LF])?;
1438        self.stream.flush()?;
1439        if self.debug {
1440            eprint!("C: {}\n", String::from_utf8(buf.to_vec()).unwrap());
1441        }
1442        Ok(())
1443    }
