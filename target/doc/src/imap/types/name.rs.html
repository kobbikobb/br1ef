1use std::borrow::Cow;
2
3/// A name that matches a `LIST` or `LSUB` command.
4#[derive(Debug, Eq, PartialEq)]
5pub struct Name {
6    // Note that none of these fields are *actually* 'static.
7    // Rather, they are tied to the lifetime of the `ZeroCopy` that contains this `Name`.
8    pub(crate) attributes: Vec<NameAttribute<'static>>,
9    pub(crate) delimiter: Option<&'static str>,
10    pub(crate) name: &'static str,
11}
12
13/// An attribute set for an IMAP name.
14#[derive(Clone, Debug, Hash, PartialEq, Eq)]
15pub enum NameAttribute<'a> {
16    /// It is not possible for any child levels of hierarchy to exist
17    /// under this name; no child levels exist now and none can be
18    /// created in the future.
19    NoInferiors,
20
21    /// It is not possible to use this name as a selectable mailbox.
22    NoSelect,
23
24    /// The mailbox has been marked "interesting" by the server; the
25    /// mailbox probably contains messages that have been added since
26    /// the last time the mailbox was selected.
27    Marked,
28
29    /// The mailbox does not contain any additional messages since the
30    /// last time the mailbox was selected.
31    Unmarked,
32
33    /// A non-standard user- or server-defined name attribute.
34    Custom(Cow<'a, str>),
35}
36
37impl NameAttribute<'static> {
38    fn system(s: &str) -> Option<Self> {
39        match s {
40            "\\Noinferiors" => Some(NameAttribute::NoInferiors),
41            "\\Noselect" => Some(NameAttribute::NoSelect),
42            "\\Marked" => Some(NameAttribute::Marked),
43            "\\Unmarked" => Some(NameAttribute::Unmarked),
44            _ => None,
45        }
46    }
47}
48
49impl<'a> From<String> for NameAttribute<'a> {
50    fn from(s: String) -> Self {
51        if let Some(f) = NameAttribute::system(&s) {
52            f
53        } else {
54            NameAttribute::Custom(Cow::Owned(s))
55        }
56    }
57}
58
59impl<'a> From<&'a str> for NameAttribute<'a> {
60    fn from(s: &'a str) -> Self {
61        if let Some(f) = NameAttribute::system(s) {
62            f
63        } else {
64            NameAttribute::Custom(Cow::Borrowed(s))
65        }
66    }
67}
68
69impl Name {
70    /// Attributes of this name.
71    pub fn attributes(&self) -> &[NameAttribute<'_>] {
72        &self.attributes[..]
73    }
74
75    /// The hierarchy delimiter is a character used to delimit levels of hierarchy in a mailbox
76    /// name.  A client can use it to create child mailboxes, and to search higher or lower levels
77    /// of naming hierarchy.  All children of a top-level hierarchy node use the same
78    /// separator character.  `None` means that no hierarchy exists; the name is a "flat" name.
79    pub fn delimiter(&self) -> Option<&str> {
80        self.delimiter
81    }
82
83    /// The name represents an unambiguous left-to-right hierarchy, and are valid for use as a
84    /// reference in `LIST` and `LSUB` commands. Unless [`NameAttribute::NoSelect`] is indicated,
85    /// the name is also valid as an argument for commands, such as `SELECT`, that accept mailbox
86    /// names.
87    pub fn name(&self) -> &str {
88        self.name
89    }
90}
