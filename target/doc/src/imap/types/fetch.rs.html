1use super::{Flag, Seq, Uid};
2use chrono::{DateTime, FixedOffset};
3use imap_proto::types::{AttributeValue, BodyStructure, Envelope, MessageSection, SectionPath};

5/// Format of Date and Time as defined RFC3501.
6/// See `date-time` element in [Formal Syntax](https://tools.ietf.org/html/rfc3501#section-9)
7/// chapter of this RFC.
8const DATE_TIME_FORMAT: &str = "%d-%b-%Y %H:%M:%S %z";
9
10/// An IMAP [`FETCH` response](https://tools.ietf.org/html/rfc3501#section-7.4.2) that contains
11/// data about a particular message. This response occurs as the result of a `FETCH` or `STORE`
12/// command, as well as by unilateral server decision (e.g., flag updates).
13#[derive(Debug, Eq, PartialEq)]
14pub struct Fetch {
15    /// The ordinal number of this message in its containing mailbox.
16    pub message: Seq,
17
18    /// A number expressing the unique identifier of the message.
19    /// Only present if `UID` was specified in the query argument to `FETCH` and the server
20    /// supports UIDs.
21    pub uid: Option<Uid>,
22
23    /// A number expressing the [RFC-2822](https://tools.ietf.org/html/rfc2822) size of the message.
24    /// Only present if `RFC822.SIZE` was specified in the query argument to `FETCH`.
25    pub size: Option<u32>,
26
27    // Note that none of these fields are *actually* 'static. Rather, they are tied to the lifetime
28    // of the `ZeroCopy` that contains this `Name`. That's also why they can't be public -- we can
29    // only return them with a lifetime tied to self.
30    pub(crate) fetch: Vec<AttributeValue<'static>>,
31    pub(crate) flags: Vec<Flag<'static>>,
32}
33
34impl Fetch {
35    /// A list of flags that are set for this message.
36    pub fn flags(&self) -> &[Flag<'_>] {
37        &self.flags[..]
38    }
39
40    /// The bytes that make up the header of this message, if `BODY[HEADER]`, `BODY.PEEK[HEADER]`,
41    /// or `RFC822.HEADER` was included in the `query` argument to `FETCH`.
42    pub fn header(&self) -> Option<&[u8]> {
43        self.fetch
44            .iter()
45            .filter_map(|av| match av {
46                AttributeValue::BodySection {
47                    section: Some(SectionPath::Full(MessageSection::Header)),
48                    data: Some(hdr),
49                    ..
50                }
51                | AttributeValue::Rfc822Header(Some(hdr)) => Some(*hdr),
52                _ => None,
53            })
54            .next()
55    }
56
57    /// The bytes that make up this message, included if `BODY[]` or `RFC822` was included in the
58    /// `query` argument to `FETCH`. The bytes SHOULD be interpreted by the client according to the
59    /// content transfer encoding, body type, and subtype.
60    pub fn body(&self) -> Option<&[u8]> {
61        self.fetch
62            .iter()
63            .filter_map(|av| match av {
64                AttributeValue::BodySection {
65                    section: None,
66                    data: Some(body),
67                    ..
68                }
69                | AttributeValue::Rfc822(Some(body)) => Some(*body),
70                _ => None,
71            })
72            .next()
73    }
74
75    /// The bytes that make up the text of this message, included if `BODY[TEXT]`, `RFC822.TEXT`,
76    /// or `BODY.PEEK[TEXT]` was included in the `query` argument to `FETCH`. The bytes SHOULD be
77    /// interpreted by the client according to the content transfer encoding, body type, and
78    /// subtype.
79    pub fn text(&self) -> Option<&[u8]> {
80        self.fetch
81            .iter()
82            .filter_map(|av| match av {
83                AttributeValue::BodySection {
84                    section: Some(SectionPath::Full(MessageSection::Text)),
85                    data: Some(body),
86                    ..
87                }
88                | AttributeValue::Rfc822Text(Some(body)) => Some(*body),
89                _ => None,
90            })
91            .next()
92    }
93
94    /// The envelope of this message, if `ENVELOPE` was included in the `query` argument to
95    /// `FETCH`. This is computed by the server by parsing the
96    /// [RFC-2822](https://tools.ietf.org/html/rfc2822) header into the component parts, defaulting
97    /// various fields as necessary.
98    ///
99    /// The full description of the format of the envelope is given in [RFC 3501 section
100    /// 7.4.2](https://tools.ietf.org/html/rfc3501#section-7.4.2).
101    pub fn envelope(&self) -> Option<&Envelope<'_>> {
102        self.fetch
103            .iter()
104            .filter_map(|av| match av {
105                AttributeValue::Envelope(env) => Some(&**env),
106                _ => None,
107            })
108            .next()
109    }
110
111    /// Extract the bytes that makes up the given `BOD[<section>]` of a `FETCH` response.
112    ///
113    /// See [section 7.4.2 of RFC 3501](https://tools.ietf.org/html/rfc3501#section-7.4.2) for
114    /// details.
115    pub fn section(&self, path: &SectionPath) -> Option<&[u8]> {
116        self.fetch
117            .iter()
118            .filter_map(|av| match av {
119                AttributeValue::BodySection {
120                    section: Some(sp),
121                    data: Some(data),
122                    ..
123                } if sp == path => Some(*data),
124                _ => None,
125            })
126            .next()
127    }
128
129    /// Extract the `INTERNALDATE` of a `FETCH` response
130    ///
131    /// See [section 2.3.3 of RFC 3501](https://tools.ietf.org/html/rfc3501#section-2.3.3) for
132    /// details.
133    pub fn internal_date(&self) -> Option<DateTime<FixedOffset>> {
134        self.fetch
135            .iter()
136            .filter_map(|av| match av {
137                AttributeValue::InternalDate(date_time) => Some(*date_time),
138                _ => None,
139            })
140            .next()
141            .and_then(
142                |date_time| match DateTime::parse_from_str(date_time, DATE_TIME_FORMAT) {
143                    Ok(date_time) => Some(date_time),
144                    Err(_) => None,
145                },
146            )
147    }
148
149    /// Extract the `BODYSTRUCTURE` of a `FETCH` response
150    ///
151    /// See [section 2.3.6 of RFC 3501](https://tools.ietf.org/html/rfc3501#section-2.3.6) for
152    /// details.
153    pub fn bodystructure<'a>(&self) -> Option<&BodyStructure<'a>> {
154        self.fetch
155            .iter()
156            .filter_map(|av| match av {
157                AttributeValue::BodyStructure(bs) => Some(bs),
158                _ => None,
159            })
160            .next()
161    }
162}
